# Code block
```cpp
#include "iostream"
#include <stdio.h>

void foo(const char* target) {
    std::cout << target << std::endl;
}

int main () {
    int x = 420;
    float y = 6.9f;
    const char* str = "Hello World";

    printf("%d\n", x);
    printf("%2.1f\n", y);

    foo(str);

    return 0;
}
```
